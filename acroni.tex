\begin{itemize}
	\item[] \textbf{IEC}: International Electrotechnical Comission Comisión Electrotécnica Internacional
	\item[] \textbf{RTOS}: Real Time Operative Sysyem, Sistema operativo en tiempo real
	\item[] \textbf{AMR}: Automatic Meter Reading, Lectura de medición automática
	\item[] \textbf{HAN}: Home Area Network, Red de área doméstica
	\item[] \textbf{NAN}: Neighbourhood Area Network, Red de área de vecindario
	\item[] \textbf{WAN}: Wide Area Network, Red de área amplia
	\item[] \textbf{ISO}: International Organization for Standardization, Organización Internacional de Normalización
	\item[] \textbf{OSI}: Open System Interconnection, Interconexión de Sistemas Abiertos
	\item[] \textbf{GUI}: Grafic User Interface, Interfaz Gráfica de Usuario	
	\item[] \textbf{NVS}: Non volatile Storage, Memoria no Volátil
	\item[] \textbf{STA}: STA, Estación (Utilizado para WiFi)
	\item[] \textbf{AP}: Access Point, Punto de Acceso
\end{itemize}
